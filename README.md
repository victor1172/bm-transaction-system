(totally_cost >= 0) NOT NULL,    order_status VARCHAR(50) CHECK (order_status IN ('Pending', 'Completed', 'Cancelled')) NOT NULL,    payment_complete_dt TIMESTAMP NULL,    modify_dt TIMESTAMP
